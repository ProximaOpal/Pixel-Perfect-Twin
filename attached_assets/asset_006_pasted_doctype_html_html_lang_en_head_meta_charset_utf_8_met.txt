<script>
(function(){
  /* ---------------------------------------------------------
     Content
     --------------------------------------------------------- */
  const SUMMARY_SECTIONS = [
    {
      eyebrow:'SECTION 1',
      title:'What is this document about?',
      body:"This upload covers the core stages of cellular respiration — how living cells convert glucose and oxygen into usable energy, carbon dioxide, and water."
    },
    {
      eyebrow:'SECTION 2',
      title:'What are the key steps covered?',
      body:"It walks through glycolysis in the cytoplasm, the citric acid cycle inside the mitochondria, and the electron transport chain that produces the bulk of the cell's ATP."
    },
    {
      eyebrow:'SECTION 3',
      title:'What should you remember?',
      body:"The big takeaway: energy production is a multi-stage relay, not a single reaction — each stage hands off molecules the next stage depends on."
    }
  ];

  const QUESTIONS = [
    {
      type:'text',
      eyebrow:'QUESTION 1',
      title:'Which stage of cellular respiration happens in the cytoplasm?',
      options:['Glycolysis', 'The citric acid cycle', 'The electron transport chain'],
      correct:0,
      submitLabel:'Continue'
    },
    {
      type:'image',
      eyebrow:'QUESTION 2',
      title:'Which diagram shows the correct process order?',
      options:[
        {blurred:true, svg:`<svg viewBox="0 0 64 64"><defs><pattern id="p0" width="8" height="8" patternTransform="rotate(45)" patternUnits="userSpaceOnUse"><rect width="8" height="8" fill="#3e4f86"/><rect width="4" height="4" fill="#5a6aa8"/><rect x="4" y="4" width="4" height="4" fill="#5a6aa8"/></pattern></defs><rect width="64" height="64" fill="url(#p0)"/></svg>`},
        {svg:`<svg viewBox="0 0 64 64"><rect width="64" height="64" fill="#dfe8f6"/><path d="M32 6c10 0 16 9 16 18 0 7-3 12-7 15l2 8H21l2-8c-4-3-7-8-7-15 0-9 6-18 16-18z" fill="#f4a259"/><path d="M32 6c10 0 16 9 16 18 0 7-3 12-7 15l2 8H32V6z" fill="#e8763a"/><rect x="27" y="49" width="10" height="8" rx="2" fill="#8a5a3c"/></svg>`},
        {svg:`<svg viewBox="0 0 64 64"><rect width="64" height="64" fill="#ffe4d6"/><path d="M32 6c10 0 16 9 16 18 0 7-3 12-7 15l2 8H21l2-8c-4-3-7-8-7-15 0-9 6-18 16-18z" fill="#f28fb2"/><path d="M22 24c0-9 6-18 10-18s10 9 10 18-4 15-10 15-10-6-10-15z" fill="#f7b6cf"/><rect x="27" y="49" width="10" height="8" rx="2" fill="#8a5a3c"/></svg>`}
      ],
      correct:2,
      submitLabel:'Continue'
    },
    {
      type:'text',
      eyebrow:'QUESTION 3',
      title:'Why does the cell need oxygen in this process?',
      options:[
        'Oxygen is the raw material converted directly into glucose',
        'Oxygen powers the citric acid cycle by replacing carbon dioxide',
        'Oxygen is the final electron acceptor in the electron transport chain, allowing ATP production to continue'
      ],
      correct:2,
      submitLabel:'Submit and show results'
    }
  ];

  /* ---------------------------------------------------------
     Elements & state
     --------------------------------------------------------- */
  const scrollArea = document.getElementById('scrollArea');
  const progressFill = document.getElementById('progressFill');
  const results = document.getElementById('results');
  const overlay = document.getElementById('transitionOverlay');

  let mode = 'summary';
  const state = {
    summary: { current:0 },
    questions: { current:0, answers:new Array(QUESTIONS.length).fill(null) }
  };

  function checkIconSVG(color){
    return `<svg viewBox="0 0 24 24" fill="none"><path d="M4 12l5 5L20 6" stroke="${color}" stroke-width="3" stroke-linecap="round" stroke-linejoin="round"/></svg>`;
  }

  function animateEntrance(){
    const entering = scrollArea.querySelectorAll('.question.enter');
    requestAnimationFrame(()=>{
      requestAnimationFrame(()=>{
        entering.forEach(sec=> sec.classList.remove('enter'));
      });
    });
  }

  /* ---------------------------------------------------------
     Summary mode — only the current section is ever rendered,
     so there is nothing to scroll. "Continue" swaps it out.
     --------------------------------------------------------- */
  function renderSummarySection(sec, index){
    const isLast = index === SUMMARY_SECTIONS.length - 1;
    return `
      <section class="question active enter" data-index="${index}">
        <p class="eyebrow">${sec.eyebrow}</p>
        <h2 class="q-title">${sec.title}</h2>
        <div class="summary-body">${sec.body}</div>
        <button class="submit-btn ready" data-continue="${index}">${isLast ? 'Finish summary' : 'Continue'}</button>
      </section>`;
  }

  function renderSummary(){
    const s = state.summary;
    scrollArea.innerHTML = renderSummarySection(SUMMARY_SECTIONS[s.current], s.current);

    const btn = scrollArea.querySelector('[data-continue]');
    if(btn){
      btn.addEventListener('click', ()=>{
        btn.classList.add('pressed');
        setTimeout(()=> btn.classList.remove('pressed'), 160);
        setTimeout(()=>{
          scrollArea.classList.add('fading');
          setTimeout(()=>{
            if(state.summary.current < SUMMARY_SECTIONS.length - 1){
              state.summary.current += 1;
              renderSummary();
            } else {
              showSummaryComplete();
            }
            scrollArea.classList.remove('fading');
          }, 260);
        }, 160);
      });
    }

    progressFill.style.width = ((s.current+1)/(SUMMARY_SECTIONS.length+1)*100) + '%';
    animateEntrance();
  }

  function showSummaryComplete(){
    results.innerHTML = `
      <svg class="medal" viewBox="0 0 24 24" fill="none"><path d="M12 2l2.4 4.9 5.4.8-3.9 3.8.9 5.4L12 14.4l-4.8 2.5.9-5.4-3.9-3.8 5.4-.8L12 2z" fill="#06c97a"/></svg>
      <p class="congrats" style="font-size:19px;">You're all caught up</p>
      <p class="score-sub">You've read through every section of this summary.</p>
      <button class="replay-btn" id="toQuestionsBtn">Practice with questions</button>
    `;
    progressFill.style.width = '100%';
    setTimeout(()=>{ results.classList.add('visible'); }, 250);
    document.getElementById('toQuestionsBtn').addEventListener('click', ()=>{
      results.classList.remove('visible');
      switchMode('questions');
    });
  }

  /* ---------------------------------------------------------
     Questions mode — same idea: one question card at a time.
     --------------------------------------------------------- */
  function renderTextQuestion(q, index){
    const selected = state.questions.answers[index];

    const optsHtml = q.options.map((opt,i)=>{
      let cls = '';
      if(selected === i){
        cls = 'selected ' + (i === q.correct ? 'correct' : 'wrong');
      }
      return `
        <div class="text-opt ${cls}" data-opt="${i}">
          ${opt}
          <span class="opt-dot">${checkIconSVG('#0894ce')}</span>
        </div>`;
    }).join('');

    const btnReady = selected !== null;
    return `
      <section class="question active enter" data-index="${index}">
        <p class="eyebrow">${q.eyebrow}</p>
        <h2 class="q-title">${q.title}</h2>
        <div class="text-options">${optsHtml}</div>
        <button class="submit-btn ${btnReady?'ready':''}" data-submit="${index}">${q.submitLabel}</button>
      </section>`;
  }

  function renderImageQuestion(q, index){
    const selected = state.questions.answers[index] !== null ? state.questions.answers[index] : 0;
    const ringClass = state.questions.answers[index] === null ? '' : (selected === q.correct ? 'correct' : 'wrong');
    const thumbs = q.options.map((opt,i)=>`
      <div class="choice ${opt.blurred?'blurred':''}" data-opt="${i}">
        <div class="thumb">${opt.svg}</div>
      </div>
    `).join('');

    const btnReady = state.questions.answers[index] !== null;
    return `
      <section class="question active enter" data-index="${index}">
        <p class="eyebrow">${q.eyebrow}</p>
        <h2 class="q-title">${q.title}</h2>
        <div class="choices" data-choices="${index}">
          <div class="sel-ring ${ringClass}" data-ring="${index}" style="transform:translateX(${selected*80}px)">
            <div class="sel-dot">${checkIconSVG('#fff')}</div>
          </div>
          ${thumbs}
        </div>
        <button class="submit-btn ${btnReady?'ready':''}" data-submit="${index}">${q.submitLabel}</button>
      </section>`;
  }

  function renderQuestionCard(q, index){
    return q.type === 'image' ? renderImageQuestion(q,index) : renderTextQuestion(q,index);
  }

  function renderQuestions(){
    const st = state.questions;
    scrollArea.innerHTML = renderQuestionCard(QUESTIONS[st.current], st.current);
    attachQuestionHandlers();
    progressFill.style.width = ((st.current+1)/(QUESTIONS.length+1)*100) + '%';
    animateEntrance();
  }

  function attachQuestionHandlers(){
    const section = scrollArea.querySelector('.question.active');
    if(!section) return;
    const index = parseInt(section.dataset.index,10);
    const q = QUESTIONS[index];

    if(q.type === 'text'){
      section.querySelectorAll('.text-opt').forEach(opt=>{
        opt.addEventListener('click', ()=>{
          state.questions.answers[index] = parseInt(opt.dataset.opt,10);
          renderQuestions();
        });
      });
    } else {
      section.querySelectorAll('.choice').forEach(choiceEl=>{
        choiceEl.addEventListener('click', ()=>{
          const i = parseInt(choiceEl.dataset.opt,10);
          state.questions.answers[index] = i;
          const ring = section.querySelector('[data-ring]');
          const isCorrect = i === q.correct;
          ring.style.transform = `translateX(${i*80}px)`;
          ring.classList.toggle('correct', isCorrect);
          ring.classList.toggle('wrong', !isCorrect);
          const btn = section.querySelector('.submit-btn');
          btn.classList.add('ready');
        });
      });
    }

    const btn = section.querySelector('.submit-btn');
    btn.addEventListener('click', ()=> handleQuestionSubmit(index));
  }

  function handleQuestionSubmit(index){
    const st = state.questions;
    if(st.answers[index] === null) return;
    const section = scrollArea.querySelector(`.question[data-index="${index}"]`);
    const btn = section.querySelector('.submit-btn');
    btn.classList.add('pressed');
    btn.disabled = true;

    setTimeout(()=>{
      btn.classList.remove('pressed');
      btn.classList.add('locked');
    }, 160);

    setTimeout(()=>{
      scrollArea.classList.add('fading');
      setTimeout(()=>{
        if(st.current < QUESTIONS.length - 1){
          st.current += 1;
          renderQuestions();
        } else {
          showQuestionsResults();
        }
        scrollArea.classList.remove('fading');
      }, 260);
    }, 600);
  }

  function showQuestionsResults(){
    let score = 0;
    QUESTIONS.forEach((q,i)=>{ if(state.questions.answers[i] === q.correct) score++; });
    const pct = Math.round((score/QUESTIONS.length)*100);

    results.innerHTML = `
      <svg class="medal" viewBox="0 0 24 24" fill="none"><path d="M12 2l2.4 4.9 5.4.8-3.9 3.8.9 5.4L12 14.4l-4.8 2.5.9-5.4-3.9-3.8 5.4-.8L12 2z" fill="#0894ce"/></svg>
      <p class="score">${pct}%</p>
      <p class="congrats">Nice work</p>
      <p class="score-sub">You got ${score} out of ${QUESTIONS.length} questions right</p>
      <button class="replay-btn" id="replayBtn">Retake questions</button>
    `;
    progressFill.style.width = '100%';
    setTimeout(()=>{ results.classList.add('visible'); }, 250);
    document.getElementById('replayBtn').addEventListener('click', ()=>{
      overlay.classList.add('run');
      setTimeout(resetQuestions, 520);
      setTimeout(()=>{ overlay.classList.remove('run'); }, 1080);
    });
  }

  function resetQuestions(){
    state.questions.current = 0;
    state.questions.answers = new Array(QUESTIONS.length).fill(null);
    results.classList.remove('visible');
    scrollArea.classList.remove('fading');
    renderQuestions();
  }

  /* ---------------------------------------------------------
     Mode toggle
     --------------------------------------------------------- */
  const modeIndicator = document.getElementById('modeIndicator');

  function renderActive(){
    if(mode === 'summary'){ renderSummary(); } else { renderQuestions(); }
  }

  function switchMode(m){
    if(m === mode) return;
    mode = m;
    document.querySelectorAll('.mode-btn').forEach(b=> b.classList.toggle('active', b.dataset.mode === m));
    modeIndicator.style.transform = m === 'questions' ? 'translateX(100%)' : 'translateX(0)';
    results.classList.remove('visible');
    scrollArea.classList.add('fading');
    setTimeout(()=>{
      renderActive();
      scrollArea.classList.remove('fading');
    }, 220);
  }

  document.querySelectorAll('.mode-btn').forEach(btn=>{
    btn.addEventListener('click', ()=> switchMode(btn.dataset.mode));
  });

  renderActive();

  /* ---------------------------------------------------------
     Upload document (bottom-left icon) -> RAG progress overlay
     --------------------------------------------------------- */
  const uploadFab = document.getElementById('uploadFab');
  const fileInput = document.getElementById('fileInput');
  const uploadToast = document.getElementById('uploadToast');
  const ragOverlay = document.getElementById('ragOverlay');
  const ragRing = document.getElementById('ragRingProgress');
  const ragPct = document.getElementById('ragPct');
  const ragLabel = document.getElementById('ragLabel');
  const ragFilename = document.getElementById('ragFilename');
  const RAG_CIRCUMFERENCE = 238.76;

  uploadFab.addEventListener('click', ()=> fileInput.click());
  fileInput.addEventListener('change', (e)=>{
    const f = e.target.files[0];
    if(!f) return;
    runRagProgress(f.name);
    fileInput.value = '';
  });

  function runRagProgress(filename){
    ragFilename.textContent = filename;
    ragLabel.textContent = 'Uploading document…';
    ragPct.textContent = '0%';
    ragRing.style.strokeDashoffset = RAG_CIRCUMFERENCE;
    ragOverlay.classList.add('show');

    let pct = 0;
    const interval = setInterval(()=>{
      pct += Math.random()*7 + 4;
      if(pct >= 100){
        pct = 100;
        clearInterval(interval);
        ragLabel.textContent = 'Done';
        updateRagVisual(pct);
        setTimeout(()=>{
          ragOverlay.classList.remove('show');
          uploadToast.textContent = 'Added to notebook: ' + filename;
          uploadToast.classList.add('show');
          setTimeout(()=> uploadToast.classList.remove('show'), 2400);
        }, 500);
        return;
      }
      if(pct > 30 && pct < 78){ ragLabel.textContent = 'Analyzing via RAG…'; }
      else if(pct >= 78){ ragLabel.textContent = 'Finalizing notebook…'; }
      updateRagVisual(pct);
    }, 180);
  }

  function updateRagVisual(pct){
    ragRing.style.strokeDashoffset = RAG_CIRCUMFERENCE * (1 - pct/100);
    ragPct.textContent = Math.round(pct) + '%';
  }

  /* ---------------------------------------------------------
     Landing -> app (auto, same wipe transition as replay)
     --------------------------------------------------------- */
  const landing = document.getElementById('landing');
  const landingOverlay = document.getElementById('landingOverlay');
  const stage = document.getElementById('stage');

  window.addEventListener('load', ()=>{
    setTimeout(()=>{
      landingOverlay.classList.add('run');
      setTimeout(()=>{
        landing.style.display = 'none';
        stage.style.opacity = '1';
        stage.style.pointerEvents = 'auto';
      }, 600);
      setTimeout(()=>{ landingOverlay.classList.remove('run'); }, 1080);
    }, 2400);
  });

  /* ---------------------------------------------------------
     Decorative green cursor — visible across the whole screen
     --------------------------------------------------------- */
  const fakeCursor = document.getElementById('fakeCursor');
  const HOVER_SELECTOR = '.text-opt, .choice, .submit-btn, .replay-btn, .mode-btn, .upload-fab, .search-bar input';

  window.addEventListener('mousemove', (e)=>{
    fakeCursor.style.transform = `translate(${e.clientX}px, ${e.clientY}px)`;
    fakeCursor.classList.toggle('hover', !!e.target.closest(HOVER_SELECTOR));
  });
  document.addEventListener('mouseleave', ()=>{
    fakeCursor.style.transform = 'translate(-999px,-999px)';
  });
  window.addEventListener('mousedown', (e)=>{
    if(e.target.closest(HOVER_SELECTOR)) fakeCursor.classList.add('press');
  });
  window.addEventListener('mouseup', ()=>{
    fakeCursor.classList.remove('press');
    fakeCursor.classList.add('click-pulse');
    setTimeout(()=> fakeCursor.classList.remove('click-pulse'), 500);
  });
})();
